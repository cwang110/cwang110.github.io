mysql常见数据类型

整型

tinyint

  • 有符号（-128 ~ 127）默认
  • 无符号 （0 ~ 255）
create table tb1(
    id int not null auto_increment parmary key,
    age tinyiny
) default charset=utf8;

create table tb2(
    id int not null auto_increment parmary key,
    age tinyint unsigned
) default charset=utf8;

int

  • 有符号（-1247483648 ~ 2147483647）
  • 无符号 （0 ~ 4294967295）

bigint

  • 有符号（-9223372036854775808 ~ 9223372036854775807）
  • 无符号 （0 ~ 18446744073709551615）

小数

float

double

decimal

准确的小数值

salary decimal(m,d),m为数字总个数（符号不算），d是小数点后的位数

create table tb3(
    id int not null primary key auto_increment,
    salary decimal(8,2)
)default charset=utf8;

insert into tb3(salary) values(1.234);
insert into tb3(salary) values(1502.55);
insert into tb3(salary) values(1024.15);

字符串

char

定长字符串

-- 固定使用11个字符串进行存储，不够11个字符也按照11个字符存储
char(11) 

-- m定标字符串的长度，最多了容纳255个字符
char(m)

create table tb4(
    id int not null auto_increment paimary key,
    mobile char(11)
)default charset=utf8;

insert into tb4(mobile) values("15713584687");

varchar

变长字符串

varchar(m)
-- m代表字符的长度，最大65535/3=最大的m

create table tb4(
    id int not null auto_increment paimary key,
    mobile char(11)
)default charset=utf8;

insert into tb4(mobile) values("15713584687");
-- 真实数据多长就按照多长存储

text

text数据类型用于保存边长的大字符串，最多65535个字符

mediumtext

longtext
