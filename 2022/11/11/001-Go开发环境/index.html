Go开发环境安装

1.下载golang for linux二进制包

curl -SL https://dl.google.com/go/go1.19.1.linux-amd64.tar.gz -o Golang.tar.gz

2.解压文件到目录

mkdir /usr/local/go
tar -zxvf golang.tar.gz -C /usr/local/

3.配置环境变量

vi /etc/profile
#在底部加入GOPATH
PATH=/usr/local/go
export PATH=$PATH:$GOPATH

source /etc/profile

#测试
go version
go version go1.19.1 linux/amd64
